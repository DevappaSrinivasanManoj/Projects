# __label__1 : negative

import pickle
import pandas as pd

with open('sentiment_model.pkl', 'rb') as f:
    loaded_tfidf, loaded_model = pickle.load(f)

# Predict
new_reviews = pd.Series(["I love it"])
X_new = loaded_tfidf.transform(new_reviews)
predictions = loaded_model.predict(X_new)
print(predictions)



new_reviews = pd.Series(["Sometimes life shows you how you don't deserve anything"])
X_new = loaded_tfidf.transform(new_reviews)
predictions = loaded_model.predict(X_new)
print(predictions)
