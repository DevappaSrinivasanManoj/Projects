import pandas as pd

new_reviews = pd.Series([
    'i like the product. but it is very expensive. everything else was satisfactory',
    'what a disappointment. my time and money would have been better spent elsewhere',
    'Purchase this without fail',
    'Never buy this'
])


results = classifier(new_reviews.tolist())

for review, result in zip(new_reviews, results):
    print(f"\nReview: {review}\nPrediction: {result['label']}, Confidence: {result['score']:.2f}")
